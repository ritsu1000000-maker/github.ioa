function submitTweet() { const input = document.getElementById('tweet-input'); const text = input.value ? input.value.trim() : ""; // 内容が空、かつメディアも添付されていない場合は投稿しない if (!text && (!attachedMediaData || !attachedMediaData.src)) { showToast("内容を入力してください"); return; } try { // 新しいポストデータを作成 const newPost = { postId: 'post_' + Date.now(), id: userId, name: myCustomInfo.name, color: myCustomInfo.color, text: text, timestamp: Date.now(), // メディアデータがあれば追加 media: (attachedMediaData && attachedMediaData.src) ? attachedMediaData : null, expireAt: Date.now() + (24 * 60 * 60 * 1000) // 24時間後に消滅 }; // 既存の投稿を取得して先頭に追加 let posts = getValidPosts(); posts.unshift(newPost); // 保存を実行 savePosts(posts); // フォームをリセット input.value = ""; attachedMediaData = { type: "", src: "" }; const mediaWrap = document.getElementById('media-preview-wrap'); if (mediaWrap) { mediaWrap.classList.remove('active'); document.getElementById('preview-content').innerHTML = ""; } // タイムラインを再描画 renderTimeline(); showToast("✅ ポストしました！"); } catch (e) { console.error("投稿エラー:", e); showToast("⚠️ 保存に失敗しました（容量がいっぱいの可能性があります）"); } }
